Khắc phục sự cố

Bạn quên Bước 2 (Thêm Service BigQuery). Hãy vào mục Services bên trái Editor, bấm dấu +, chọn BigQuery API.
Tên Dataset trong code không khớp với tên bạn tạo trên BigQuery. Kiểm tra lại chính tả.
Code.gs 
// ==========================================
// ISC DATA TEAM - AUTOMATED UPLOADER v2.1
// ==========================================

function onOpen() {
  // Tạo menu trên thanh công cụ Sheet
  SpreadsheetApp.getUi()
    .createMenu('🚀 ISC BigQuery Tools')
    .addItem('Tải lên BigQuery', 'uploadToBigQuery')
    .addToUi();
}

function uploadToBigQuery() {
  // --- CẤU HÌNH (SỬA Ở ĐÂY) ---
  const PROJECT_ID = 'isc-data-platform'; 
  const DATASET_ID = 'marketing_analytics'; // <-- Sửa thành tên Dataset của bạn
  // ---------------------------

  const sheet = SpreadsheetApp.getActiveSheet();
  const data = sheet.getDataRange().getValues();
  
  // 1. Tự động làm sạch tên bảng (chỉ lấy chữ thường và số)
  let tableName = sheet.getName().toLowerCase().replace(/[^a-z0-9]/g, "_");
  
  const headers = data[0];
  const rows = data.slice(1);
  
  if (rows.length === 0) {
    Browser.msgBox("❌ Lỗi: Sheet trống.");
    return;
  }

  // 2. Tạo Schema (Mặc định là STRING để tránh lỗi)
  const schema = headers.map(header => ({
    name: header.toLowerCase().replace(/\s+/g, "_"),
    type: 'STRING'
  }));

  // 3. Chuyển đổi CSV an toàn (Xử lý dấu phẩy)
  const csvData = rows.map(row => 
    row.map(cell => {
      let cellStr = cell.toString();
      if (cellStr.search(/[,"\n]/) !== -1) {
        return '"' + cellStr.replace(/"/g, '""') + '"'; 
      }
      return cellStr;
    }).join(",")
  ).join("\n");

  const blob = Utilities.newBlob(csvData, 'application/octet-stream');

  // 4. Cấu hình Job Upload
  const jobSpec = {
    configuration: {
      load: {
        destinationTable: {
          projectId: PROJECT_ID,
          datasetId: DATASET_ID,
          tableId: tableName
        },
        schema: { fields: schema },
        writeDisposition: 'WRITE_TRUNCATE', // Ghi đè toàn bộ
        sourceFormat: 'CSV',
        allowQuotedNewlines: true
      }
    }
  };

  try {
    const job = BigQuery.Jobs.insert(jobSpec, PROJECT_ID, blob);
    const jobId = job.jobReference.jobId;
    
    // 5. Chờ kết quả (Polling)
    let status = 'RUNNING';
    let attempts = 0;
    
    while (status === 'RUNNING' || status === 'PENDING') {
      Utilities.sleep(1000); 
      const checkJob = BigQuery.Jobs.get(PROJECT_ID, jobId);
      status = checkJob.status.state;
      
      if (status === 'DONE') {
        if (checkJob.status.errorResult) {
          Browser.msgBox("❌ Thất bại: " + checkJob.status.errorResult.message);
        } else {
          Browser.msgBox("✅ Thành công! Table: " + tableName);
        }
      }
      attempts++;
      if (attempts > 20) break;
    }
    
  } catch (e) {
    Browser.msgBox("❌ Lỗi Script: " + e.toString());
  }
}
