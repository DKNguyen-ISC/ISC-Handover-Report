Data Engineer's "Pro Tips"

The Date Format Trap

BigQuery is strict about dates. It prefers YYYY-MM-DD.

❌ 25/09/2025 (DD/MM/YYYY)
❌ September 25, 2025
✅ 2025-09-25

Beware "Ghost" Spaces

"Stationery " is not the same as "Stationery". Trailing spaces break database joins.

Tip: Use =TRIM(A1) in Sheets to clean text columns before uploading.

Colors & Styles Don't Upload

Highlighting a cell Red or Bold does not transfer to BigQuery. It only cares about the raw value.

If a product is "Urgent", add a column named "status" and type "urgent" in it.

Final Sanity Check

  • Row 1 contains only **English, snake_case headers**.
  • **No merged cells** anywhere in the data range.
  • Dates are in **YYYY-MM-DD** format.
  • Numbers are **raw values** (no currency/symbols).

// ==========================================
// ISC DATA TEAM - AUTOMATED UPLOADER v2.0 (Robust)
// ==========================================

function onOpen() {
  // Adds a custom menu to your Google Sheet
  SpreadsheetApp.getUi()
    .createMenu('🚀 ISC BigQuery Tools')
    .addItem('Upload Current Sheet', 'uploadToBigQuery')
    .addToUi();
}

function uploadToBigQuery() {
  // --- CONFIGURATION (EDIT THIS SECTION ONLY) ---
  const PROJECT_ID = 'isc-data-platform'; // Your Project ID
  const DATASET_ID = 'marketing_analytics'; // Your Dataset ID
  // --------------------------------------------

  const sheet = SpreadsheetApp.getActiveSheet();
  const data = sheet.getDataRange().getValues();
  
  // 1. Create Table Name from Sheet Name (Clean it up)
  let tableName = sheet.getName().toLowerCase().replace(/[^a-z0-9]/g, "_");
  
  // 2. Prepare Data (Headers + Rows)
  const headers = data[0];
  const rows = data.slice(1);
  
  if (rows.length === 0) {
    Browser.msgBox("❌ Error: Sheet is empty (no data rows).");
    return;
  }

  // 3. Create Schema (Safest Mode: All Fields = STRING)
  const schema = headers.map(header => ({
    name: header.toLowerCase().replace(/\s+/g, "_"),
    type: 'STRING'
  }));

  // 4. Prepare CSV payload (FIXED: Handles commas properly)
  const csvData = rows.map(row => 
    row.map(cell => {
      let cellStr = cell.toString();
      // If cell has comma, newline, or quote, wrap it in quotes
      if (cellStr.search(/[,"\n]/) !== -1) {
        return '"' + cellStr.replace(/"/g, '""') + '"'; 
      }
      return cellStr;
    }).join(",")
  ).join("\n");

  const blob = Utilities.newBlob(csvData, 'application/octet-stream');

  // 5. Configure Job (WRITE_TRUNCATE = Replace table)
  const jobSpec = {
    configuration: {
      load: {
        destinationTable: {
          projectId: PROJECT_ID,
          datasetId: DATASET_ID,
          tableId: tableName
        },
        schema: { fields: schema },
        writeDisposition: 'WRITE_TRUNCATE', // Replaces old data
        sourceFormat: 'CSV',
        allowQuotedNewlines: true // Allow multiline text
      }
    }
  };

  // 6. Send and MONITOR the Job
  try {
    const job = BigQuery.Jobs.insert(jobSpec, PROJECT_ID, blob);
    const jobId = job.jobReference.jobId;
    
    // --- NEW: WAIT FOR COMPLETION ---
    let status = 'RUNNING';
    let attempts = 0;
    
    while (status === 'RUNNING' || status === 'PENDING') {
      Utilities.sleep(1000); // Wait 1 second
      const checkJob = BigQuery.Jobs.get(PROJECT_ID, jobId);
      status = checkJob.status.state;
      
      if (status === 'DONE') {
        if (checkJob.status.errorResult) {
          // Found a REAL error
          Browser.msgBox("❌ Job Failed: " + checkJob.status.errorResult.message);
          Logger.log(checkJob.status.errors); 
        } else {
          // Success!
          Browser.msgBox("✅ Success! Data uploaded to table: " + tableName);
        }
      }
      
      attempts++;
      if (attempts > 20) { 
         Browser.msgBox("⚠️ Job is taking too long. Check BigQuery Console.");
         break;
      }
    }
    
  } catch (e) {
    Browser.msgBox("❌ Script Error: " + e.toString());
  }
}

How to bypass it:
